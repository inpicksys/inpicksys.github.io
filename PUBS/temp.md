    public class RouletteItem
    {
        public string Label { get; set; } = "";
        public double Weight { get; set; }
    }

    private List<RouletteItem> Items = new()
    {
        new RouletteItem { Label = "A", Weight = 1 },
        new RouletteItem { Label = "B", Weight = 2 },
        new RouletteItem { Label = "C", Weight = 3 }
    };

    private string newLabel = "";
    private double newWeight = 1;
    private RouletteItem? SelectedItem;

    private void AddItem()
    {
        if (!string.IsNullOrWhiteSpace(newLabel) && newWeight > 0)
        {
            Items.Add(new RouletteItem { Label = newLabel, Weight = newWeight });
            newLabel = "";
            newWeight = 1;
        }
    }

    private void Spin()
    {
        var total = Items.Sum(i => i.Weight);
        var r = Random.Shared.NextDouble() * total;
        double acc = 0;
        foreach (var item in Items)
        {
            acc += item.Weight;
            if (r <= acc)
            {
                SelectedItem = item;
                return;
            }
        }
    }

    private double TotalWeight => Items.Sum(i => i.Weight);

    private string GetArcPath(int index)
    {
        double startAngle = GetStartAngle(index);
        double endAngle = startAngle + GetAngle(index);
        double r = 150;

        double x1 = r * Math.Cos(startAngle);
        double y1 = r * Math.Sin(startAngle);
        double x2 = r * Math.Cos(endAngle);
        double y2 = r * Math.Sin(endAngle);

        var largeArc = (endAngle - startAngle) > Math.PI ? 1 : 0;

        return $"M0,0 L{x1:F2},{y1:F2} A{r},{r} 0 {largeArc},1 {x2:F2},{y2:F2} Z";
    }

    private double GetAngle(int index)
    {
        return (Items[index].Weight / TotalWeight) * 2 * Math.PI;
    }

    private double GetStartAngle(int index)
    {
        return Items.Take(index).Sum(item => (item.Weight / TotalWeight) * 2 * Math.PI);
    }

    private string GetColor(int index)
    {
        var hue = (index * 360.0 / Items.Count) % 360;
        return $"hsl({hue},70%,50%)";
    }

    private double GetTextX(int index)
    {
        double midAngle = GetStartAngle(index) + GetAngle(index) / 2;
        return 100 * Math.Cos(midAngle);
    }

    private double GetTextY(int index)
    {
        double midAngle = GetStartAngle(index) + GetAngle(index) / 2;
        return 100 * Math.Sin(midAngle);
    }
